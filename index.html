    <!-- Importazione fogli di stile e script di Leaflet.js -->

    <link rel="stylesheet" href="https://unpkg.com/leaflet@1.9.4/dist/leaflet.css" />

    <script src="https://unpkg.com/leaflet@1.9.4/dist/leaflet.js"></script>


    <style>

        body {

            margin: 0;

            padding: 0;

            font-family: Arial, sans-serif;

        }

        #map {

            width: 100vw;

            height: 100vh;

        }

    </style>

</head>

<body>


    <!-- Contenitore della mappa -->

    <div id="map"></div>


    <script>

        // Inizializzazione della mappa centrata sull'Europa

        const map = L.map('map').setView([45.0, 10.0], 4);


        // Aggiunta del layer cartografico di base (OpenStreetMap)

        L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {

            maxZoom: 18,

            attribution: '© OpenStreetMap contributors'

        }).addTo(map);


        // Dizionario per memorizzare i marker delle navi tramite il loro MMSI (identificativo univoco)

        const ships = {};


        // Inizializzazione della connessione WebSocket verso AISStream

        const socket = new WebSocket("wss://stream.aisstream.io/v0/stream");


        socket.onopen = function() {

            console.log("Connessione WebSocket stabilita con successo.");

            

            // Creazione del payload di sottoscrizione

            const subscriptionMessage = {

                APIKey: "INSERIRE_QUI_LA_PROPRIA_API_KEY", // Sostituire con la chiave API reale

                BoundingBoxes: [[[-90, -180], [90, 180]]], // Copertura globale

                FiltersShipMMSI: [], 

                FilterMessageTypes: ["PositionReport"] // Richiesta esclusiva dei dati di posizione

            };

            

            // Invio della richiesta al server

            socket.send(JSON.stringify(subscriptionMessage));

        };


        socket.onmessage = function(event) {

            // Decodifica del messaggio in ingresso

            const aisMessage = JSON.parse(event.data);

            

            if (aisMessage.MessageType === "PositionReport") {

                const report = aisMessage.Message.PositionReport;

                const mmsi = report.UserID;

                const lat = report.Latitude;

                const lon = report.Longitude;


                // Validazione delle coordinate (prevenzione di dati corrotti)

                if (lat >= -90 && lat <= 90 && lon >= -180 && lon <= 180) {

                    

                    if (ships[mmsi]) {

                        // Aggiornamento della posizione se la nave è già tracciata

                        ships[mmsi].setLatLng([lat, lon]);

                    } else {

                        // Creazione di un nuovo marker se la nave non è presente

                        const marker = L.circleMarker([lat, lon], {

                            radius: 3,

                            color: 'blue',

                            fillColor: '#3388ff',

                            fillOpacity: 0.8

                        }).addTo(map);

                        

                        // Aggiunta di un tooltip con l'identificativo

                        marker.bindTooltip("MMSI: " + mmsi);

                        ships[mmsi] = marker;

                    }

                }

            }

        };


        socket.onerror = function(error) {

            console.error("Errore nella connessione WebSocket:", error);

        };


        socket.onclose = function() {

            console.warn("Connessione WebSocket terminata.");

        };

    </script>

</body>
